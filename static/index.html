What's happening under the hood

// Client-side: reading the SSE stream with fetch()
const res = await fetch("/chat/stream", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ message })
});

// The response body is a ReadableStream
const reader = res.body.getReader();
const decoder = new TextDecoder();

while (true) {
  const { done, value } = await reader.read();  // blocks until next chunk
  if (done) break;

  const chunk = decoder.decode(value);
  // chunk looks like: "data: {\"token\": \"Hello \"}\n\n"

  for (const line of chunk.split("\n")) {
    if (!line.startsWith("data: ")) continue;
    const payload = line.slice(6);
    if (payload === "[DONE]") return;

    const { token } = JSON.parse(payload);
    appendToChat(token);  // immediately visible to user!
  }
}
